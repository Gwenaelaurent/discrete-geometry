\section*{Writing}

\begin{enumerate}
\item Let $P$ be a $3$-dimensional polytope with $n$
  vertices. 
  Prove or disprove: 
  If the graph of~$P$ is complete (i.e., between every pair of points
  there is an edge on the convex hull), then $n=4$, so that $P$~is a tetrahedron.
\end{enumerate}
